# Создаём меню
При клике на пункты меню переходим к секциям по якорям на главной:
- делаем переход от ссылки к заголовку секции плавным
- при переходе не точно попадаем в координаты, потому что хотелось бы чтобы при переходе оставался зазор между заголовком и верхней границей браузера
- делаем меню фиксированным, однако в этом случаем координаты при переходе еще больше сбиваются

Делаем на JavаScript:
- плавный скрол
- скрол идет до нужной точки (в нужные координаты)
- пункт меню по которым был клик подсвечивается, становится активным
- при этом при скроле страницы колесом мыши, пункты меню подсвечиваются сами по себе

Схема создания кода:
- подписываемся на клик по ссылкам меню, используем функцию delegate(),

Пояснения к коду:

Код `document.querySelector(this.hash)` ищет элемент на странице, который соответствует значению хеш-фрагмента (hash) текущего URL. Этот метод используется для навигации по странице, когда пользователь переходит по внутренним ссылкам, содержащим хеш-фрагменты. Например, если у вас есть ссылка `<a href="#section-1">Перейти к разделу 1</a>`, то при нажатии на неё, если страница уже загружена, произойдёт перемещение к элементу с идентификатором `section-1`, который может быть определён как `<div id="section-1">`.

    window.scrollTo({

    });

    <script>
    window.addEventListener('load', function() {
      // Получаем доступ к меню
      let menu = document.querySelector('.menu');

      // С помощью функции delegate() получаем доступ к ссылкам меню
      delegate(menu, 'a', 'click', function(e) {
        e.preventDefault(); // отменяем клик по ссылке

        // получаем доступ к якорю у ссылки
        let target = document.querySelector(this.hash); // указывает на элемент страницы у которого id равен хэшу страницы, в нашем случае это заголовки
        scrollToElem(target); // отправляем заголовок в эту функцию и вычисляем его координату y
        setActiveMenuItem(menu, this);
      });

      // Прямой переход по ссылке с хэшем, там где > проверяем имеется ли у ссылки хэш
      // Без кода ниже, при прямом переходе, заголовок будет скрываться меню, не будет (-70)
      let hash = window.location.hash;
      let autoTarget = hash.length > 0 ? document.querySelector(hash) : null;
      // console.log(autoTarget);

      if(autoTarget !== null) {
        scrollToElem(autoTarget);
        setActiveMenuItem(menu, menu.querySelector(`[href$="${hash}"]`));
        // menu.querySelector(`[href$="${hash}"]`) - чтобы подсветить ссылку при переходе из вне
      }
    });

    // Функция delegate()
    function delegate(box, selector, eventName, handler) {
      box.addEventListener(eventName, function(e) {
        let elem = e.target.closest(selector);

        if(elem !== null && box.contains(elem)) {
          handler.call(elem, e);
        }
      });
    }

    // Подсвечиваем активные пункты меню
    function setActiveMenuItem(menu, item) {
      menu.querySelectorAll('a').forEach(link => link.classList.remove('menu__link-active'));
      item.classList.add('menu__link-active');
    }

    // Перемещаемся к заголовку, предварительно вычислив его координату Y
    function scrollToElem(el) {
      let top = el.offsetTop - 70;

      window.scrollTo({
        top,
        behavior: "smooth"
      });
    }
    </script>
